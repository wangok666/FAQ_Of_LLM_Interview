## 1. 合并2个有序整数数组 / 合并2个序链表

```python
def merge(nums1, m, nums2, n):
    """
    假设 nums1 的长度为 m，nums2 的长度为 n

    1. 初始化三个指针：
        - p1 指向 nums1 的末尾（m-1）
        - p2 指向 nums2 的末尾（n-1）
        - p 指向合并后数组的末尾（m+n-1）
    2. 比较与填充：
        比较 nums1[p1] 和 nums2[p2]，将较大的那个数放入 nums1[p]
        向前移动相应的指针（p 和选中的 p1 或 p2）
    3. 处理剩余元素:
        如果 nums2 还有剩余，直接将其拷贝到 nums1 的前面
    """
    # 初始化指针
    p1 = m - 1
    p2 = n - 1
    p = m + n - 1

    # 从后向前遍历
    while p1 >= 0 and p2 >= 0:
        if nums1[p1] > nums2[p2]:
            nums1[p] = nums1[p1]
            p1 -= 1
        else:
            nums1[p] = nums2[p2]
            p2 -= 1
        p -= 1

    # 如果 nums2 还有剩余
    nums1[: p2 + 1] = nums2[: p2 + 1]


class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next


def mergeTwoLists(l1, l2):
    # 创建哨兵节点
    dummy = ListNode(-1)
    curr = dummy

    while l1 and l2:
        if l1.val <= l2.val:
            curr.next = l1
            l1 = l1.next
        else:
            curr.next = l2
            l2 = l2.next
        curr = curr.next

    # 连接剩余的链表部分
    curr.next = l1 if l1 else l2

    # 返回真正的头节点
    return dummy.next
```

```python
def findLength(nums1, nums2):
    """
    假设 `dp[i][j]` 表示以 `nums1[i-1]` 和 `nums2[j-1]` 结尾的公共子数组的长度：
    * 如果 `nums1[i-1] == nums2[j-1]`，那么这个位置的公共长度就是前一个位置的长度加 1：
        $$dp[i][j] = dp[i-1][j-1] + 1$$
    * 如果两者不相等，则以它们结尾的公共子数组长度为 0：
        $$dp[i][j] = 0$$
    """

    m, n = len(nums1), len(nums2)
    # 初始化 DP 表，大小为 (m+1) x (n+1)
    dp = [[0] * (n + 1) for _ in range(m + 1)]
    max_len = 0

    for i in range(1, m + 1):
        for j in range(1, n + 1):
            # 如果当前元素相等
            if nums1[i - 1] == nums2[j - 1]:
                dp[i][j] = dp[i - 1][j - 1] + 1
                # 更新全局最大长度
                max_len = max(max_len, dp[i][j])
            # 不相等时 dp[i][j] 默认为 0，无需额外操作

    return max_len


if __name__ == "__main__":

    n2 = [3, 2, 1, 4, 7]
    # 最长公共子数组是 [3, 2, 1]
    print(f"最长公共子数组长度: {findLength(n1, n2)}")
```

```python
def longestCommonSubsequence(text1: str, text2: str) -> int:
    """
    我们定义 $dp[i][j]$ 为 `text1` 的前 $i$ 个字符和 `text2` 的前 $j$ 个字符的最长公共子序列长度。

    * **情况 A：字符相等 (`text1[i-1] == text2[j-1]`)**
        找到了一个公共字符，长度加 1：
        $$dp[i][j] = dp[i-1][j-1] + 1$$
    * **情况 B：字符不相等**
        既然当前字符不匹配，我们就看：是“舍弃 `text1` 的当前字符”结果更好，还是“舍弃 `text2` 的当前字符”结果更好？取两者的最大值：
        $$dp[i][j] = \max(dp[i-1][j], dp[i][j-1])$$
    """
    m, n = len(text1), len(text2)
    # 初始化 dp 矩阵
    dp = [[0] * (n + 1) for _ in range(m + 1)]

    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if text1[i - 1] == text2[j - 1]:
                # 字符匹配，左上方结果 + 1
                dp[i][j] = dp[i - 1][j - 1] + 1
            else:
                # 字符不匹配，取左边或上边的最大值
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1])

    return dp[m][n]


if __name__ == "__main__":
    t1, t2 = "abcde", "ace"
    print(f"最长公共子序列长度: {longestCommonSubsequence(t1, t2)}")
```


## 3. 最长回文子串

```python
def longestPalindrome(s: str) -> str:
    """
    假设 `dp[i][j]` 表示字符串从索引 `i` 到 `j` 的子串是否为回文串（布尔值）：
    * **状态转移方程：**
        如果 `s[i] == s[j]` 且满足以下任一条件，则 `dp[i][j] = True`：
        1. 子串长度小于等于 2（即 `j - i < 2`）。
        2. 内部子串也是回文，即 `dp[i+1][j-1]` 为真。
        $$dp[i][j] = (s[i] == s[j]) \text{ and } (j - i < 3 \text{ or } dp[i+1][j-1])$$
    * **基本情况：**
        单个字符始终是回文，即 $$dp[i][i] = True$$。
    """

    n = len(s)
    if n < 2:
        return s

    # 初始化 DP 表，大小为 n x n
    dp = [[False] * n for _ in range(n)]
    # 每个单独的字符都是回文
    for i in range(n):
        dp[i][i] = True

    max_len = 1
    begin = 0

    # 这里的遍历顺序很重要：要保证计算 dp[i][j] 时，其内部状态 dp[i+1][j-1] 已被计算
    # 因此采用“由短到长”的子串长度遍历方式
    for length in range(2, n + 1):  # 子串长度从 2 到 n
        for i in range(n):  # 左边界 i
            j = i + length - 1  # 右边界 j

            if j >= n:
                break

            if s[i] == s[j]:
                if j - i < 3:
                    dp[i][j] = True
                else:
                    dp[i][j] = dp[i + 1][j - 1]
            else:
                dp[i][j] = False

            # 只要 dp[i][j] 为真，表示 s[i..j] 是回文，记录并更新最大长度
            if dp[i][j] and (j - i + 1) > max_len:
                max_len = j - i + 1
                begin = i

    return s[begin : begin + max_len]


if __name__ == "__main__":
    test_str = "babad"
    # "babad" 的最长回文子串是 "bab" 或 "aba"
    print(f"原字符串: {test_str}")
    print(f"最长回文子串: {longestPalindrome(test_str)}")
```

## 4. 二分法查找

```python
def binarySearch(nums, target):
    """
    二分查找的前提是数组必须是有序的（升序）。
    假设查找范围为 `[left, right]`：
    * 计算中间索引：
        $$mid = left + (right - left) // 2$$
    * 如果 `nums[mid] == target`，说明找到了，返回 `mid`。
    * 如果 `nums[mid] < target`，说明目标值在右侧，更新左边界：
        $$left = mid + 1$$
    * 如果 `nums[mid] > target`，说明目标值在左侧，更新右边界：
        $$right = mid - 1$$
    """

    left, right = 0, len(nums) - 1

    while left <= right:
        # 防止溢出的写法，等同于 (left + right) // 2
        mid = left + (right - left) // 2

        if nums[mid] == target:
            # 找到目标值
            return mid
        elif nums[mid] < target:
            # 目标在右半部分
            left = mid + 1
        else:
            # 目标在左半部分
            right = mid - 1

    # 未找到目标值
    return -1


if __name__ == "__main__":
    nums = [1, 3, 5, 7, 9, 11, 13, 15]
    target = 7

    result = binarySearch(nums, target)

    if result != -1:
        print(f"数字 {target} 在数组中的索引为: {result}")
    else:
        print(f"数组中不存在数字 {target}")

    # 测试不存在的情况
    target_not_exist = 10
    print(f"数字 {target_not_exist} 的查找结果: {binarySearch(nums, target_not_exist)}")
```

## 5. 最小合法子串

```python
def longestValidParentheses(s: str) -> int:
    """
    假设 `dp[i]` 表示以字符 `s[i]` 结尾的最长有效括号子串的长度：
    * 如果 `s[i] == '('`，以左括号结尾的子串一定不是有效的，所以 `dp[i] = 0`。
    * 如果 `s[i] == ')'`，需要检查前一个字符：
        1. 若 `s[i-1] == '('`，形式如 `...()`，则：
           $$dp[i] = dp[i-2] + 2$$
        2. 若 `s[i-1] == ')'` 且 `s[i - dp[i-1] - 1] == '('`，形式如 `...((...))`，则：
           $$dp[i] = dp[i-1] + 2 + dp[i - dp[i-1] - 2]$$
    """
    n = len(s)
    if n == 0:
        return 0

    # 初始化 DP 数组，长度为 n，初始值为 0
    dp = [0] * n
    max_len = 0

    for i in range(1, n):
        if s[i] == ")":
            # 场景 1: 形如 "......()"
            if s[i - 1] == "(":
                dp[i] = (dp[i - 2] if i >= 2 else 0) + 2
            # 场景 2: 形如 "......))"
            # 需要找到与当前 ')' 匹配的左括号位置：i - dp[i-1] - 1
            elif i - dp[i - 1] > 0 and s[i - dp[i - 1] - 1] == "(":
                # 基本长度 = 内部有效长度 dp[i-1] + 当前匹配的一对 (2)
                # 还要加上该匹配对之前的有效长度 dp[i - dp[i-1] - 2]
                prev_len = dp[i - dp[i - 1] - 2] if (i - dp[i - 1] >= 2) else 0
                dp[i] = dp[i - 1] + 2 + prev_len

            # 更新全局最大长度
            max_len = max(max_len, dp[i])

    return max_len


if __name__ == "__main__":
    s1 = "(()"
    s2 = ")()())"
    s3 = "()(())"

    print(f"输入: '{s1}' -> 最长有效括号长度: {longestValidParentheses(s1)}")
    print(f"输入: '{s2}' -> 最长有效括号长度: {longestValidParentheses(s2)}")
    print(f"输入: '{s3}' -> 最长有效括号长度: {longestValidParentheses(s3)}")
```

## 6. 最长前缀

```python
def longestCommonPrefix(strs):
    """
    假设 `LCP(S1, ..., Sn)` 是字符串集的前缀函数：
    * 基本思想：
        $$LCP(S_1 \dots S_n) = LCP(LCP(LCP(S_1, S_2), S_3), \dots S_n)$$
    * 逻辑步骤：
        1. 检查数组是否为空。
        2. 以第一个字符串为基准，逐个字符向后比较。
        3. 如果遇到字符不匹配或到达某个字符串末尾，则截断当前前缀。
    """

    if not strs:
        return ""

    # 取第一个字符串作为初始公共前缀
    prefix = strs[0]
    count = len(strs)

    for i in range(1, count):
        # 逐个比较后续字符串，不断更新 prefix
        j = 0
        while j < len(prefix) and j < len(strs[i]) and prefix[j] == strs[i][j]:
            j += 1

        # 截取当前最长的公共部分
        prefix = prefix[:j]

        # 如果中途公共前缀变为空字符串，提前结束
        if not prefix:
            break

    return prefix


if __name__ == "__main__":
    test_cases = [
        ["flower", "flow", "flight"],
        ["dog", "racecar", "car"],
        ["apple", "app", "april"],
    ]

    for s in test_cases:
        result = longestCommonPrefix(s)
        print(f"字符串列表 {s} 的最长公共前缀是: '{result}'")
```

```python
def fibonacci(n):
    """
    假设 `dp[i]` 表示斐波那契数列中第 `i` 个位置的数值：
    * 状态转移方程：当前数值等于前两个数值之和：
        $$dp[i] = dp[i-1] + dp[i-2]$$
    * 边界条件（基础情况）：
        $$dp[0] = 0, \quad dp[1] = 1$$
    """

    if n <= 1:
        return n

    # 初始化 DP 数组，大小为 n + 1
    dp = [0] * (n + 1)

    # 设置初始状态
    dp[0] = 0
    dp[1] = 1

    # 从下标 2 开始计算，直到 n
    for i in range(2, n + 1):
        # 根据状态转移方程更新
        dp[i] = dp[i - 1] + dp[i - 2]

    return dp[n]

    result = fibonacci(n)
    print(f"斐波那契数列第 {n} 个数是: {result}")
```

## 8. 最大和的连续子数组

```python
def maxSubArray(nums):
    """
    假设 `dp[i]` 表示以 `nums[i]` 结尾的连续子数组的最大和：
    * 如果 `dp[i-1] > 0`，则当前元素加入前面的子数组可以使和更大：
        $$dp[i] = dp[i-1] + nums[i]$$
    * 如果 `dp[i-1] <= 0`，则前面的子数组对当前和没有增益，应从当前元素重新开始计算：
        $$dp[i] = nums[i]$$
    * 状态转移方程简化为：
        $$dp[i] = \max(dp[i-1] + nums[i], nums[i])$$
    """

    if not nums:
        return 0

    n = len(nums)
    # 初始化 DP 表，dp[i] 记录以第 i 个元素结尾的最大子数组和
    dp = [0] * n
    dp[0] = nums[0]

    # 结果初始化为第一个元素
    max_sum = dp[0]

    for i in range(1, n):
        # 决定是“接龙”还是“另起炉灶”
        dp[i] = max(dp[i - 1] + nums[i], nums[i])
        # 每一轮迭代后更新全局最大和
        max_sum = max(max_sum, dp[i])

    return max_sum


if __name__ == "__main__":
    nums = [-2, 1, -3, 4, -1, 2, 1, -5, 4]
    # 最大连续子数组是 [4, -1, 2, 1]，和为 6
    print(f"最大子数组和: {maxSubArray(nums)}")
```
